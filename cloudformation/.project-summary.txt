QUICK START COMMAND:
  cd cloudformation
  ./deploy.sh

  OR manually:
  aws cloudformation create-stack \
    --stack-name cloud-optimizer \
    --template-body file://cloud-optimizer-standalone.yaml \
    --parameters file://parameters/trial.json \
    --capabilities CAPABILITY_NAMED_IAM \
    --region us-east-1

PARAMETERS REQUIRED:
  - DBPassword: 8-41 alphanumeric characters
  - JWTSecretKey: Minimum 32 characters

  Generate with:
  openssl rand -base64 32 | tr -dc 'a-zA-Z0-9' | head -c 32

OUTPUTS PROVIDED:
  - ApplicationURL: Load balancer DNS
  - DatabaseEndpoint: RDS endpoint
  - ECSCluster: Cluster name
  - ECSService: Service name
  - VPCId: VPC identifier

POST-DEPLOYMENT:
